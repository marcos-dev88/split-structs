type Envelope struct {
	XMLName xml.Name `xml:"Envelope"`
	Text    string   `xml:",chardata"`
	Soapenv string   `xml:"soapenv,attr"`
	Head    string   `xml:"head,attr"`
	Soap    string   `xml:"soap,attr"`
	Header  struct {
		Text   string `xml:",chardata"`
		Header struct {
			Text        string `xml:",chardata"`
			ClientID    string `xml:"client-id"`
			Service     string `xml:"service"`
			MessageType string `xml:"message-type"`
			Paradigm    string `xml:"paradigm"`
			MessageID   string `xml:"message-id"`
			Params      struct {
				Text  string `xml:",chardata"`
				Param struct {
					Text  string `xml:",chardata"`
					Name  string `xml:"name"`
					Value string `xml:"value"`
				} `xml:"param"`
			} `xml:"params"`
		} `xml:"header"`
	} `xml:"Header"`
	Body struct {
		Text    string `xml:",chardata"`
		Request struct {
			Text             string `xml:",chardata"`
			Paradigm         string `xml:"paradigm"`
			SmsTimestamp     string `xml:"sms-timestamp"`
			SubStatus        string `xml:"sub-status"`
			TemplateID       string `xml:"template-id"`
			Title            string `xml:"title"`
			InvoiceNumberTwo string `xml:"invoice-number-two"`
			Activity         struct {
				Text   string `xml:",chardata"`
				Date   string `xml:"date"`
				ID     string `xml:"id"`
				Result string `xml:"result"`
			} `xml:"activity"`
			Customer struct {
				Text         string `xml:",chardata"`
				Msisdn       string `xml:"msisdn"`
				SocialSecNo  string `xml:"social-sec-no"`
				Cnpj         string `xml:"cnpj"`
				Name         string `xml:"name"`
				Birthdate    string `xml:"birthdate"`
				Type         string `xml:"type"`
				Password     string `xml:"password"`
				Segment      string `xml:"segment"`
				SystemOrigin string `xml:"system-origin"`
				Address      struct {
					Text         string `xml:",chardata"`
					Country      string `xml:"country"`
					Municipality string `xml:"municipality"`
					Name         string `xml:"name"`
					Number       string `xml:"number"`
					State        string `xml:"state"`
					Street       string `xml:"street"`
					ZipCode      string `xml:"zip-code"`
				} `xml:"address"`
				AddressTwo struct {
					Text            string `xml:",chardata"`
					CountryTwo      string `xml:"country-two"`
					MunicipalityTwo string `xml:"municipality-two"`
					NumberTwo       string `xml:"number-two"`
					StateTwo        string `xml:"state-two"`
					StreetTwo       string `xml:"street-two"`
					ZipCodeTwo      string `xml:"zip-code-two"`
				} `xml:"address-two"`
				Device struct {
					Text  string `xml:",chardata"`
					Brand string `xml:"brand"`
					Code  string `xml:"code"`
					Imei  string `xml:"imei"`
					Model string `xml:"model"`
					Price string `xml:"price"`
				} `xml:"device"`
				Invoice struct {
					Text    string `xml:",chardata"`
					DueDate string `xml:"due-date"`
					Number  string `xml:"number"`
				} `xml:"invoice"`
				Plan struct {
					Text            string `xml:",chardata"`
					ActivationDate  string `xml:"activation-date"`
					CancelationType string `xml:"cancelation-type"`
					CurrentPlan     string `xml:"current-plan"`
					DestinyPlan     string `xml:"destiny-plan"`
					AmountPlan      string `xml:"amount-plan"`
					EndCycle        string `xml:"end-cycle"`
					Name            string `xml:"name"`
					StartCycle      string `xml:"start-cycle"`
				} `xml:"plan"`
				Telephone struct {
					Text    string `xml:",chardata"`
					Contact string `xml:"contact"`
				} `xml:"telephone"`
				Loyalty struct {
					Text  string `xml:",chardata"`
					Value struct {
						Text  string `xml:",chardata"`
						One   string `xml:"one"`
						Two   string `xml:"two"`
						Three string `xml:"three"`
						Four  string `xml:"four"`
						Five  string `xml:"five"`
						Six   string `xml:"six"`
						Seven string `xml:"seven"`
						Eight string `xml:"eight"`
					} `xml:"value"`
					Plan struct {
						Text  string `xml:",chardata"`
						One   string `xml:"one"`
						Two   string `xml:"two"`
						Three string `xml:"three"`
					} `xml:"plan"`
					Date struct {
						Text  string `xml:",chardata"`
						One   string `xml:"one"`
						Two   string `xml:"two"`
						Three string `xml:"three"`
						Four  string `xml:"four"`
						Five  string `xml:"five"`
						Six   string `xml:"six"`
					} `xml:"date"`
				} `xml:"loyalty"`
			} `xml:"customer"`
			Date struct {
				Text     string `xml:",chardata"`
				Emission string `xml:"emission"`
			} `xml:"date"`
			Interaction struct {
				Text           string `xml:",chardata"`
				CommercialName string `xml:"commercial-name"`
				Creation       string `xml:"creation"`
				Note           string `xml:"note"`
				ReasonOne      string `xml:"reason-one"`
				ReasonTwo      string `xml:"reason-two"`
				ReasonThree    string `xml:"reason-three"`
			} `xml:"interaction"`
			Invoice struct {
				Text           string `xml:",chardata"`
				AutomaticDebit struct {
					Text string `xml:",chardata"`
					Code string `xml:"code"`
				} `xml:"automatic-debit"`
				BankSlipNumber string `xml:"bank-slip-number"`
				Barcode        string `xml:"barcode"`
				Payment        struct {
					Text      string `xml:",chardata"`
					Balance   string `xml:"balance"`
					LimitDate string `xml:"limit-date"`
				} `xml:"payment"`
				ReferenceMonth string `xml:"reference-month"`
				URL            string `xml:"url"`
			} `xml:"invoice"`
			Protocol struct {
				Text   string `xml:",chardata"`
				Date   string `xml:"date"`
				Number string `xml:"number"`
			} `xml:"protocol"`
			RelatedServiceRequest struct {
				Text         string `xml:",chardata"`
				Channel      string `xml:"channel"`
				CreationDate string `xml:"creation-date"`
				Customer     struct {
					Text string `xml:",chardata"`
					Name string `xml:"name"`
				} `xml:"customer"`
				Interaction struct {
					Text           string `xml:",chardata"`
					CommercialName string `xml:"commercial-name"`
					ReasonOne      string `xml:"reason-one"`
					ReasonTwo      string `xml:"reason-two"`
					ReasonThree    string `xml:"reason-three"`
				} `xml:"interaction"`
				Number   string `xml:"number"`
				Protocol struct {
					Text      string `xml:",chardata"`
					StartDate string `xml:"start-date"`
					EndDate   string `xml:"end-date"`
					Number    string `xml:"number"`
				} `xml:"protocol"`
				Size     string `xml:"size"`
				Response struct {
					Text    string `xml:",chardata"`
					Channel string `xml:"channel"`
					Date    string `xml:"date"`
				} `xml:"response"`
				ServiceRequest struct {
					Text         string `xml:",chardata"`
					Channel      string `xml:"channel"`
					CreationDate string `xml:"creation-date"`
					Number       string `xml:"number"`
				} `xml:"service-request"`
			} `xml:"related-service-request"`
		} `xml:"request"`
	} `xml:"Body"`
}
